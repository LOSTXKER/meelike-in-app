// src/middleware.ts
import { NextResponse } from 'next/server';
import type { NextRequest } from 'next/server';

export function middleware(request: NextRequest) {
  const hostname = request.headers.get('host') || '';
  const pathname = request.nextUrl.pathname;
  
  // Extract subdomain
  // hostname: john.meelike.com or www.meelike.com or meelike.com
  const subdomain = hostname.split('.')[0];
  
  // Skip main domain and www
  const mainDomains = ['www', 'meelike', 'localhost'];
  
  if (subdomain && !mainDomains.includes(subdomain)) {
    // Rewrite to store route
    // john.meelike.com/checkout â†’ /store/john/checkout
    const newUrl = new URL(`/store/${subdomain}${pathname}`, request.url);
    return NextResponse.rewrite(newUrl);
  }
  
  return NextResponse.next();
}

export const config = {
  matcher: [
    // Skip internal paths
    '/((?!api|_next/static|_next/image|favicon.ico).*)',
  ],
};
